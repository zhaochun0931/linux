Most modern Linux distributions actually use a new-and-improved daemon called rsyslog. rsyslog is capable of forwarding logs to remote servers.





#install
  
add-apt-repository ppa:adiscon/v8-stable
apt-get update
apt-get install rsyslog


#check rsyslog version
rsyslogd -v






/etc/services



/etc/rsyslog.conf
/etc/rsyslog.d/50-default.conf


# specify the port in the /etc/rsyslog.conf

$ModLoad imudp
$UDPServerRun 514


systemctl restart rsyslog.service
systemctl status rsyslog.service


ss -antlup


# test to send the message to syslog

echo "hello rsyslog" > /dev/udp/localhost/514


logger "hello rsyslog"
