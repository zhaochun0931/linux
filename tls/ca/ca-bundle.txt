# ca
openssl genrsa -out ca.key
openssl req -new -x509 -days 365 -key ca.key -out ca.crt







# server
openssl genrsa -out server.key 4096
openssl req -new -key server.key -out server.csr

openssl x509 -req -in server.csr -CA ca.crt -CAkey ca.key -CAcreateserial -out server.crt
openssl x509 -req -in server.csr -CA ca.crt -CAkey ca.key -CAcreateserial -out server.crt -sha256 -extfile server_cert_ext.cnf









# client
openssl genrsa -out client.key 4096
openssl req -new -key client.key -out client.csr
openssl x509 -req -in client.csr -CA ca.crt -CAkey ca.key -CAcreateserial -out client.cert -sha256 -extfile client_cert_ext.cnf
