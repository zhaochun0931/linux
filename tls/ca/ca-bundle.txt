# ca
openssl genrsa -out ca.key
openssl rsa -noout -text -in ca.key


openssl req -new -x509 -days 365 -key ca.key -out ca.cert
openssl x509 -noout -text -in ca.cert







# server
openssl genrsa -out server.key 4096
openssl req -new -key server.key -out server.csr
openssl rsa -noout -text -in server.key
openssl req -noout -text -in server.csr


openssl x509 -req -days 365 -in server.csr -CA ca.cert -CAkey ca.key -CAcreateserial -out server.crt
openssl x509 -noout -text -in server.crt







# client
openssl genrsa -out client.key 4096
openssl req -new -key client.key -out client.csr
openssl x509 -req -in client.csr -CA ca.cert -CAkey ca.key -out client.cert -CAcreateserial -days 365 -sha256 -extfile client_cert_ext.cnf

openssl rsa -noout -text -in client.key
openssl req -noout -text -in client.csr
openssl x509 -noout -text -in client.cert
