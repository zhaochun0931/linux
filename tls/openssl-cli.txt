The genrsa command is a more legacy way to generate RSA private keys. The private key generated by genrsa is usually in the traditional PEM format.

openssl genrsa -out private.key







The genpkey command is a more general-purpose tool that is part of OpenSSL's newer command set and is more flexible and modern than genrsa







openssl enc -list


--- encrypt and decrypt with the same cipher (password)
--- encrypt and decrypt with the public key and private key



# encrpytion
openssl enc -e -aes-128-cbc -pbkdf2 -in /etc/passwd -out file2

# decryption
openssl enc -d -aes-128-cbc -pbkdf2 -in file2 -out file3

















# generate a private key
openssl genpkey -algorithm RSA -out my-key.key

# extrace a public key from the private key
openssl pkey -in key.pem -pubout -out public.key



openssl genrsa -out demo.key
openssl rsa -pubout -in demo.key -out demo.pub






# encrypt with the public key
openssl pkeyutl -encrypt -in demo.txt -out demo.txt.enc -pubin -inkey demo.pub

# decrypt with the private key
openssl pkeyutl -decrypt -inkey demo.key -in demo.txt.enc -out test.out.txt
