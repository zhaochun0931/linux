3. Basic BGP Configuration (eBGP)
Assuming r1 is 172.27.0.2 (AS 65001) and r2 is 172.27.0.3 (AS 65002)



docker exec -it r1 bash


ip addr add 10.1.1.1/24 dev lo


vtysh



configure terminal


! 1. Define your local AS number
router bgp 65001

no bgp ebgp-requires-policy


  ! 2. Identify your neighbor's IP and their AS number
  neighbor 172.27.0.3 remote-as 65002
  
  ! 3. Enter the IPv4 address family to activate the peer
  address-family ipv4 unicast
    ! Tell the neighbor about a network you "own"
    network 10.1.1.0/24
  exit-address-family
exit
