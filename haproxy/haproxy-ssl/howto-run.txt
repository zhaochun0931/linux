mkdir certs

openssl req -x509 -nodes -newkey rsa:2048 \
  -keyout certs/server.key \
  -out certs/server.crt \
  -days 365 \
  -subj "/CN=localhost"


cat certs/server.crt certs/server.key > certs/server.pem







docker compose up

curl --cacert certs/server.crt https://localhost
