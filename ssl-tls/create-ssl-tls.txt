# generate rootCA
openssl genrsa -out rootCA.key 4096
openssl req -x509 -new -nodes -key rootCA.key -sha256 -days 1024 -out rootCA.crt -subj "/C=CN/ST=Shanghai/O=ORG/OU=ORG_UNIT/CN=www.rootCA.com"



# generate server key
openssl genrsa -out mydomain.key 2048

openssl req -new -sha256 -key mydomain.key \
-subj "/C=CN/ST=Shanghai/O=ORG/OU=ORG_UNIT/CN=www.xiaoming.com" \
-reqexts SAN \
-config <(cat /etc/ssl/openssl.cnf <(printf "\n[SAN]\nsubjectAltName=DNS:www.xiaoming.com")) \
-out mydomain.csr


# sign with the rootCA

openssl x509 -req -extfile <(printf "subjectAltName=DNS:www.xiaoming.com") -days 120 \
-in mydomain.csr \
-CA rootCA.crt -CAkey rootCA.key -CAcreateserial \
-out mydomain.crt -sha256







# export to pfx
openssl pkcs12 -export -in rootCA.crt -inkey rootCA.key -out rootCA.pfx


# add the pfx to the trusted root certification authorities


# add the entry for the www.xiaoming.com in the /etc/hosts or windows hosts files
1.1.1.1 www.xiaoming.com


# successfully visit https://www.xiaoming.com from the browser




openssl x509 -in mydomain.crt -text -noout


openssl verify -CAfile rootCA.crt mydomain.crt






# how to verify the certificate

openssl s_server -cert mydomain.crt -key mydomain.key -accept 8443

openssl s_client -connect localhost:8443 -CAfile rootCA.crt
